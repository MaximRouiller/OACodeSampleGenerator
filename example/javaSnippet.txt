// Deployments_CreateOrUpdateAtScope

HttpClient client = HttpClient.newHttpClient();

HttpRequest request = HttpRequest.newBuilder()
  .uri(URI.create("https://managemement.azure.com/{scope}/providers/Microsoft.Resources/deployments/{deploymentName}?api-version=2021-04-01"))
  .header("Content-Type", "application/json")
  .PUT(BodyPublishers.ofFile(Paths.get("body.json")))
  .build();

HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
System.out.println(response.statusCode());
System.out.println(response.body());
